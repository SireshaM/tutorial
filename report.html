            function grade(){
            var sub1=parseInt(document.getElementById("sub1").value);
            var sub2=parseInt(document.getElementById("sub2").value);
            var sub3=parseInt(document.getElementById("sub3").value);
            var result ="";
            if (sub1 > 34 && sub2 > 34 && sub3 > 34)
            {
                var total =(sub1+sub2+sub3);
                var avg =Math.round(total/3);
                if(avg>=75)
                {
                    result="Congratulations!! The student passed with Distinction!!";
                }
                else if(avg >=60)
                {
                    result ="The student passed in First Class!!";
                }
                else if (avg>=50){
                    result="The student passed in Second Class!!";
                }
                else if(avg>=40) {
                    result="The student passed in Third Class!!";
                }
                else{
                    result="Enter valid Details";
                }
            document.getElementById("totavg").innerHTML="The total marks are "+total+" and average is "+avg;
            }
            else{
                result="The student has Failed!!";
            }
            document.getElementById("result").innerHTML=result;           
            }
